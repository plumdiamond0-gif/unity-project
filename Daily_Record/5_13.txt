using System;
using System.Collections;
using UnityEngine;
using UnityEngine.SceneManagement;

public class SceneLoadManager : MonoBehaviour
{
    private AssetManager _assetManager;

    public string NextSceneName { get; private set; }
    public string CurrentSceneName => SceneManager.GetActiveScene().name;
    //현재 활성화된 씬의 이름 가져옴
    public void Init(AssetManager assetManager)
    {
        _assetManager = assetManager;
    }

    public void NextLoadScene(string nextSceneName, Action OnSceneCompleted)
    {
        if (string.IsNullOrEmpty(nextSceneName))
        {
            Debug.LogError("nextSceneName IsNullOrEmpty");
            return;
        }

        NextSceneName = nextSceneName;

        StartCoroutine(ProcessEmptyScene(OnSceneCompleted));
    }


    private IEnumerator ProcessEmptyScene(Action OnSceneCompleted)
    {
        yield return unloadUnusedAssets();

        AsyncOperation asyncOperation = SceneManager.LoadSceneAsync("Scene_Empty");
//asyncOperation : 지금 씬 로딩 작업 진행 중임
을 나타내는 객체.


        yield return asyncOperation;
//asyncOperation 이 끝날 때까지 코루틴 일시 정지
//while (!asyncOperation.isDone)
{
yield return null;
}

        yield return GlobalCallback.SetWaitUntil(
            () =>
            {
                return asyncOperation.isDone;
            },
            () =>
            {
                StartCoroutine(ProcessNextScene(OnSceneCompleted));
            }
        );
    }

    private IEnumerator ProcessNextScene(Action OnSceneCompleted)
    {
        if (string.IsNullOrEmpty(NextSceneName))
        {
            Debug.LogError("nextSceneName IsNullOrEmpty");
            yield break;
        }

        if (_assetManager == null)
        {
            yield break;
        }

        _assetManager.LoadScene(NextSceneName, OnSceneCompleted);

        yield return unloadUnusedAssets();
    }

    private IEnumerator unloadUnusedAssets()
    {
        AsyncOperation async = Resources.UnloadUnusedAssets();
        yield return async;

        GC.Collect();
        GC.Collect();
        GC.Collect();
        GC.Collect();
        GC.Collect();
    }
}

using UnityEngine;
using System.Collections;
using System.Collections.Generic;
using System;

public class GlobalCallback : MonoBehaviour

{
    private static readonly Dictionary<float, WaitForSeconds> _dicWaitForSeconds = new Dictionary<float, WaitForSeconds>();
//new WaitForSeconds(1f) 를 계속 만들면 GC가 생길 수 있음.
//그래서 한 번 만든 WaitForSeconds(1f)를 저장해두고 재사용.

    public static readonly WaitForEndOfFrame WaitForEndOfFrame = new WaitForEndOfFrame();


    public static WaitForSeconds WaitForSecond(float sec)
    {
        if(_dicWaitForSeconds.TryGetValue(sec, out WaitForSeconds result) == false)
        {
result = new WaitForSeconds(sec);
            _dicWaitForSeconds.Add(sec, result);
        }
        return result;
    }
//받은 float이 딕셔너리에 이미 키값으로 있는지 확인 후 없으면 result에 WaitForSeconds 넣어주고 딕셔너리에 추가.
//만약 딕셔너리에 새로운 밸류 넣고 키값으로 딕셔너리에서 밸류 찾아서 result에 넣으면 비효율적.
//C#에서 객체는 복사되는 게 아니라 참조를 들고 있기에 어차피 생성되는 객체는 하나.
//딕셔너리는 지금 호출을 위한 게 아니라 미래 호출 재사용을 위해 캐시하는 것임.


    public static IEnumerator SetTimeOut(float sec, Action callback)
    {
        yield return WaitForSecond(sec);
//여기서 sec를 넘겨주고 그만큼 기다림.
//WaitForSecond(sec)가 넘겨주는 Result는 WaitForSeconds 객체

        callback?.Invoke();
    }

    public static IEnumerator SetEndOffFrame(Action callback)
    {
        yield return WaitForEndOfFrame;
        callback?.Invoke();
    }

    public static IEnumerator SetWaitUntil(Func<bool> predicate, Action callback)
//Func<bool> : 조건 함수 
/*bool 함수()
{
    return asyncOperationHandle.IsDone;
}
*/
    {
        yield return new WaitUntil(predicate);
        callback?.Invoke();
//왜 조건 함수로 받냐? : WaitUntil 자체가 지금 당장의 bool 값이 아니라 매 프레임마다 다시 검사할 조건식이 필요하기 때문.

    }
}
//SetTimeOut이 단순히 일정 시간동안만 기다리는 거라면 SetWaitUntil은 어떠한 조건식이 참이 될 때까지 기디림.




using System;
using System.Collections;
using System.Collections.Generic;
using UnityEngine;
using UnityEngine.AddressableAssets;
using UnityEngine.ResourceManagement.AsyncOperations;

public class AssetManager : MonoBehaviour
{
    private readonly Dictionary<string, AsyncOperationHandle> _dicHadles = new Dictionary<string, AsyncOperationHandle>();

    public void LoadAsset<T>(string key, Action<T> callback)
    {
        StartCoroutine(ProcessLoad(key, callback));
    }

    public void LoadScene(string key, Action callback)
    {
        AsyncOperationHandle handle = Addressables.LoadSceneAsync(key);
        handle.Completed += (item) => {
            callback?.Invoke();
        };
    }

    private IEnumerator ProcessLoad<T>(string key, Action<T> callback)
    {
        if(_dicHadles.TryGetValue(key, out AsyncOperationHandle asyncOperationHandle))
//asyncOperationHandle는 Addressables.LoadAssetAsync가 이미 실행 중인 작업을 가리키는 핸들, 
//밑에서 실행한 바로 그 로딩이 끝날 때까지 기다리기
        {
            yield return GlobalCallback.SetWaitUntil(() =>
            {
                return asyncOperationHandle.IsDone;
            },
            () =>
            {
                Debug.Log($"리소스 로드 완료: {key}");
                callback?.Invoke((T)asyncOperationHandle.Result);
//같은 에셋을 쓰더라도 서로 다른 행동(callback)을 할 수 있기에 Invoke 따로 또 함.
            });
		yield break;
        }
//로드하려는 에셋의 키값을 지닌 밸류가 이미 있다는 건 이미 이전에 누가 로드를 시작했다는 뜻
//여기서는 새로 로드 안 하고 기존에 진행 중인 로드 기다림.


        AsyncOperationHandle handle = Addressables.LoadAssetAsync<T>(key);
//에셋 로딩 작업을 시작하는 코드
//handle은 로딩 작업 자체를 관리하는 객체
//로딩이 완료되면 handle.Result에 실제로 로딩된 에셋이 들어 있음.(GameObject)

        _dicHadles[key] = handle;

        yield return handle;
        
        if(handle.Status == AsyncOperationStatus.Succeeded)
        {
            Debug.Log($"리소스 로드 완료: {key}");
            callback?.Invoke((T)handle.Result);
//UIManager에서 넘겨줬던 callback함수 실행하면서 매개변수로  handle.Result를 넣어라
//자동으로 go 자리에 내가 로드한 에셋(GameObject 타입)이 들어감.
//go는 로드된 프리팹 그 자체
//callback 자체가 GameObject 타입이기에 매개변수로 꼭 GameObject 넘겨줘야 함.
        }
        else
        {
            Debug.LogError($"데이터 로드 실패: {key}");
        }
    }

    public void Release(string key)
    {
        if (_dicHadles.TryGetValue(key, out AsyncOperationHandle asyncOperationHandle))
        {
            Addressables.Release(asyncOperationHandle);
            _dicHadles.Remove(key);
        }

    }

    public void ReleaseAll()
    {
        foreach (var item in _dicHadles)
        {
            Addressables.Release(item);
        }

        _dicHadles.Clear();
    }
}
