IEnumerator FadeIn()
{
canvasGroup.alpha = 0;
//처음에는 투명
float time = 0;

while (time < fadeTime)
{
time += Time.deltaTime;

canvasGroup.alpha = time / fadeTime;

yield return null;
}

canvasGroup.alpha = 1;
//마지막에 완전히 선명하게.
}
