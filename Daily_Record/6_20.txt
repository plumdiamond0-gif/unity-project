using System;
using System.Collections;
using UnityEngine;
using UnityEngine.AI;
using UnityEngine.Rendering.Universal;

[CreateAssetMenu(menuName = "Weapon/Effects/Stun")]


public class StunEffect : ScriptableObject, IWeaponEffect
{
    [SerializeField] float stunTime;
    public void Apply(GameObject target, float level)
    {
        EnemyMovement enemy = target.GetComponent<EnemyMovement>();
        if (enemy != null)
        {
            float finalStunTime  = stunTime * Mathf.Pow(1.15f, level);
            enemy.beStun(finalStunTime);
            Debug.Log("stun");
        }
    }
}

여기서 슬로우 이펙트를 여러 무기들이 사용하고 있음. 
그리고 그 무기들마다 기본 슬로우 어마운트, 타임이 다름. 
레벨에 따른 증가량은 모든 무기가 같아서 paw로 처리했다 해도
무기마다 다른 베이스 슬로우 데이터를 처리하려면?


현재 플레이어가 쓰는 무기 데이터까지 가져오는 건 부적절


SlowEffect = “효과 로직 (공통 규칙)”
무기 = “베이스 수치 (slowTime, slowAmount)”
level = “공통 성장 공식”

[SerializeField] float slowTime;
[SerializeField] float slowAmount;
이게 사실상 “무기별 베이스 데이터” 역할이야.
즉:
무기 A → SlowEffect (0.5 / 30%)
무기 B → SlowEffect (1.2 / 10%)
무기 C → SlowEffect (2.0 / 50%
이미 ScriptableObject 하나로 “무기별 차이”를 처리할 수 있는 구조


최종 구조 :
1) SlowEffect = “템플릿”
slowTime, slowAmount = 무기 기본값
level = 성장값
2) 무기마다 ScriptableObject 하나씩 만들기
